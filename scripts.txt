<script>

const params = new URLSearchParams(window.location.search)

const tv = params.get("tv")

let atual = ""

async function atualizar(){

 try{

  let r = await fetch("/painel/state.json?t=" + Date.now())

  let estado = await r.json()

  if(!tv || !estado[tv]) return

  let pagina = estado[tv]

  let novaPagina = "/painel/layouts/" + pagina

  if(atual !== novaPagina){

   document.getElementById("frame").src = novaPagina
   atual = novaPagina

  }

 }catch(e){
  console.log("Erro ao atualizar tela",e)
 }

}

setInterval(atualizar,2000)

atualizar()

</script>

</body>

</html>













UPDATE-SERVER.JS:

const express = require("express")
const fs = require("fs")

const app = express()

app.use(express.json())

const STATE_FILE = "/var/www/html/painel/state.json"


app.post("/update",(req,res)=>{

 let {tv,pagina} = req.body

 let state = JSON.parse(fs.readFileSync(STATE_FILE))

 state[tv] = pagina

 fs.writeFileSync(STATE_FILE,JSON.stringify(state,null,2))

 res.send({status:"ok"})

})

app.listen(3000,"0.0.0.0",()=>{
 console.log("Servidor rodando na porta 3000")
})
 console.log("Servidor de atualização rodando")
